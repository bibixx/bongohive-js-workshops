## switch

``` js
let roadLight = "red"

switch(roadLight) {
    case "greed":
        console.log("drive safely");
        break;
    case "yellow":
        console.log("please stop");
        break;
    case "red":
        console.log("STOP!");
        break;
    default:
        console.error("Unknown light value, please fix!");
}
```
